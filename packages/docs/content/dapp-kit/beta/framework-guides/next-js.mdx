DApp Kit SDK works seamlessly with Next.js through the React bindings.

## Installation

```sh npm2yarn
npm i  @mysten/dapp-kit-react @mysten/dapp-kit-core
```

## Setup

### 1. Create dApp Kit Instance

```ts
// app/dapp-kit.ts
import { createDAppKit } from '@mysten/dapp-kit-react';
import { getFullnodeUrl, SuiClient } from '@mysten/sui/client';

export const dAppKit = createDAppKit({
	enableBurnerWallet: process.env.NODE_ENV === 'development',
	networks: ['mainnet', 'testnet'],
	defaultNetwork: 'testnet',
	createClient(network) {
		return new SuiClient({ network, url: getFullnodeUrl(network) });
	},
});

// global type registration necessary for the hooks to work correctly
declare module '@mysten/dapp-kit-react' {
	interface Register {
		dAppKit: typeof dAppKit;
	}
}
```

### 2. Handle SSR with Dynamic Imports

```tsx
// app/ClientOnlyConnectButton.tsx
import { DAppKitProvider, ConnectButton } from '@mysten/dapp-kit-react';
import { dAppKit } from './dapp-kit.ts';

export default function ClientOnlyConnectButton() {
	return (
		<DAppKitProvider dAppKit={dAppKit}>
			<ConnectButton />
		</DAppKitProvider>
	);
}
```

```tsx
// app/page.tsx
'use client';

import dynamic from 'next/dynamic';

const ClientApp = dynamic(() => import('./ClientOnlyConnectButton'), {
	ssr: false,
	loading: () => <div>ClientOnlyConnectButton is not rendered on server</div>,
});

export default function Home() {
	return <ClientApp />;
}
```
